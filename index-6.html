Copy-pasteable code snippets for implementing the typewriter animation in your Wix environment. See typewriter-wix-prompt.md for full documentation.

Enable Dev Mode → select your Text element → set ID to typewriterText → paste in Page tab: 

// Wix Velo — Typewriter Effect
const WORDS = [
  "your static hand for saliva ejection.",
  "your hands-free dental assistant alternative.",
  "maximum clinical efficiency and hygiene."
];
const TYPING_SPEED   = 80;
const PAUSE_DURATION = 2000;

$w.onReady(function () {
  let wordIndex = 0;
  let charIndex = 0;

  function typeWord() {
    const word = WORDS[wordIndex];
    if (charIndex <= word.length) {
      $w("#typewriterText").text = word.substring(0, charIndex) + "|";
      charIndex++;
      setTimeout(typeWord, TYPING_SPEED);
    } else {
      setTimeout(() => {
        charIndex = 0;
        wordIndex = (wordIndex + 1) % WORDS.length;
        typeWord();
      }, PAUSE_DURATION);
    }
  }

  typeWord();
});
